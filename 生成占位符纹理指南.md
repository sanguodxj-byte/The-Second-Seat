### 安装 ImageMagick
```powershell
# 使用 Chocolatey 安装
choco install imagemagick

# 或下载：https://imagemagick.org/
```

### 生成占位符
```powershell
cd "Textures\UI"

# Ready - 灰白色 256x256
magick -size 256x256 xc:#CCCCCC NarratorButton_Ready.png
```

## ?? PowerShell 脚本生成

### 使用 .NET System.Drawing
```powershell
# 生成占位符纹理.ps1
Add-Type -AssemblyName System.Drawing

$outputDir = "Textures\UI"
if (-not (Test-Path $outputDir)) {
    New-Item -Path $outputDir -ItemType Directory -Force
}

$textures = @(
    @{Name="NarratorButton_Ready.png"; Color="204,204,204"},
    @{Name="NarratorButton_Processing.png"; Color="255,184,77"},
    @{Name="NarratorButton_Error.png"; Color="255,51,51"},
    @{Name="NarratorButton_Disabled.png"; Color="128,128,128"}
)

foreach ($texture in $textures) {
    $bitmap = New-Object System.Drawing.Bitmap(256, 256)
    $graphics = [System.Drawing.Graphics]::FromImage($bitmap)
    
    $rgb = $texture.Color -split ','
    $color = [System.Drawing.Color]::FromArgb(255, $rgb[0], $rgb[1], $rgb[2])
    $brush = New-Object System.Drawing.SolidBrush($color)
    
    $graphics.FillRectangle($brush, 0, 0, 256, 256)
    
    $path = Join-Path $outputDir $texture.Name
    $bitmap.Save($path, [System.Drawing.Imaging.ImageFormat]::Png)
    
    Write-Host "? 已生成：$($texture.Name)" -ForegroundColor Green
    
    $graphics.Dispose()
    $bitmap.Dispose()
}

Write-Host ""
Write-Host "?? 占位符生成完成！" -ForegroundColor Magenta
Write-Host ""
Write-Host "下一步：" -ForegroundColor Yellow
Write-Host "  1. 运行 .\一键部署.ps1" -ForegroundColor White
Write-Host "  2. 启动游戏测试" -ForegroundColor White
Write-Host "  3. 用真实设计替换占位符" -ForegroundColor White
```
